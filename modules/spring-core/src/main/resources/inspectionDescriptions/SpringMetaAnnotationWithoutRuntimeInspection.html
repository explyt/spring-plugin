Reports wrong retention policy for Spring meta-annotation.

Example: (omitted @Retention)


@Target({ElementType.TYPE, ElementType.METHOD})
@Profile("test")
@interface MyProfile { // reports "Spring meta-annotation should have RUNTIME RetentionPolicy"
    @AliasFor(annotation = Profile.class, value = "value")
    String[] profile();
}

Example: (wrong retention policy)


@Target({ElementType.TYPE, ElementType.METHOD})
@Retention(RetentionPolicy.SOURCE)
@Profile("test")
@interface MyProfile { // reports "Spring meta-annotation should have RUNTIME RetentionPolicy"
    @AliasFor(annotation = Profile.class, value = "value")
    String[] profile();
}
