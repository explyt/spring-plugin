Reports in the META-INFO/additional-spring-configuration-metadata.json configuration file:
  • duplicate value
  • unresolved references
  • missing and deprecated properties
  • invalid values
Example:

  {
    "properties": [
      {
        "name": "main.name", // 'Duplicate for 'main.name''
        "type": "java.lang.String"
      },
      {
        "name": "main.name", // 'Duplicate for 'main.name''
        "type": "java.lang.String"
      },
      {
        // 'Add required 'name''
        "type": "java.lang.String"
      },
      {
        "name": "main.foo.foo2.max",
        "type": "java.lang.Double",
        "description": "Max in double" // 'Text should end with '.''
      },
      {
        "name": "main.max-sessions-per-connection",
        "type": "java.lang.Integer",
        "description": "Maximum number connections ",
        "sourceType": "com.boot.Unknown", // 'Cannot resolve class 'Unknown''
        "defaultValue": 1,
        "deprecated": true // 'The property 'deprecated''
      },
    ],
    "hints": [
      {
        "name": "main.contexts1.keys", // 'Cannot resolve configuration key reference 'main.contexts1''
        "values": [
          {
            "value": "sample1"
          },
          {
            "value": "sample2"
          }
        ]
      },
      {
        "name": "main.mime-type",
        "providers": [
          {
            "name": "handle-as", // 'Add required parameters: target'
            "parameters": {
            }
          }
        ]
      }
    ]
  }
